### PDF Text Extraction

```js
const pdfParse = require("pdf-parse");

async function extractTextFromPDF(buffer) {
  const data = await pdfParse(buffer);
  if (!data.text || data.text.trim().length === 0) {
    throw new Error("Scanned PDF detected");
  }
  return data.text;
}
```

### Transaction Parsing (Foundation)

```js
function parseTransactions(rawText) {
  const lines = rawText.split("\n").map(l => l.trim()).filter(Boolean);
  const transactions = [];

  for (const line of lines) {
    const dateMatch = line.match(/\d{2}\/\d{2}\/\d{4}/);
    const amounts = line.match(/[\d,]+\.\d{2}/g);

    if (!dateMatch || !amounts || amounts.length < 2) continue;

    const credit = parseFloat(amounts[1].replace(/,/g, ""));
    if (credit <= 0) continue;

    transactions.push({
      date: dateMatch[0],
      credit,
      raw_description: line
    });
  }
  return transactions;
}
```

### Generate Excel

```js
const XLSX = require("xlsx");

function generateExcel(transactions) {
  const data = [["Date", "Description", "Credit"]];
  transactions.forEach(t =>
    data.push([t.date, t.raw_description, t.credit])
  );

  const wb = XLSX.utils.book_new();
  const ws = XLSX.utils.aoa_to_sheet(data);
  XLSX.utils.book_append_sheet(wb, ws, "Transactions");

  return XLSX.write(wb, { type: "buffer", bookType: "xlsx" });
}
```

### Upload Endpoint

```js
app.post("/upload-statement", upload.single("file"), async (req, res) => {
  try {
    const file = req.file;
    if (!file) return res.status(400).json({ error: "No file uploaded" });

    if (file.mimetype === "application/pdf") {
      const text = await extractTextFromPDF(file.buffer);
      const txs = parseTransactions(text);
      const excel = generateExcel(txs);
      return res.json({ message: "PDF converted", count: txs.length });
    }

    if (file.mimetype.includes("spreadsheet")) {
      return res.json({ message: "Excel received" });
    }

    res.status(400).json({ error: "Unsupported file type" });
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
});
```
